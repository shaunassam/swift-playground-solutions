func navigateAroundWall() {
    if isBlockedRight {
        moveForward()
    } else {
        turnRight()
        moveForward()
    }
    if isBlocked && isBlockedRight {
            turnLeft()
            moveForward()
    } 
}
    
while !isOnClosedSwitch {
    navigateAroundWall()
    if isOnGem {
        collectGem()
        turnLeft()
        turnLeft()
    }
    if isOnClosedSwitch {
        toggleSwitch()
        break
    }
}
